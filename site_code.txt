        // URL PARAMETER PARSING FOR WEBSITE TEMPLATE
        function parseURLParameters() {
            const hash = window.location.hash.substring(1);
            if (!hash) return;

            const params = hash.split(',');
            console.log('Website Template Parameters received:', params);

            if (params.length >= 16) {
                const [
                    primary, secondary, accent, sectionBg1, sectionBg2, sectionBg3, sectionBg4, companyName, heroTitle, heroSubtitle,
                    agentName, agentTitle, agentIframeUrl, productTitle, productDescription, finalCtaTitle
                ] = params;

                // Update CSS variables for colors
                const root = document.documentElement;
                root.style.setProperty('--primary-color', `#${primary}`);
                root.style.setProperty('--secondary-color', `#${secondary}`);
                root.style.setProperty('--accent-color', `#${accent}`);

                // Add these to dynamically set section backgrounds
                // Använd faktiska färgparametrar från URL:en:
                root.style.setProperty('--agent-section-bg', `#${sectionBg1}`);
                root.style.setProperty('--product-section-bg', `#${sectionBg2}`);
                root.style.setProperty('--capabilities-section-bg', `#${sectionBg3}`);
                root.style.setProperty('--social-section-bg', `#${sectionBg4}`);

                // Update meta theme color
                //document.querySelector('meta[name="theme-color"]').content = `#${primary}`;

                // Decode text content
                const decodedCompanyName = decodeURIComponent(companyName.replace(/\+/g, ' '));
                const decodedHeroTitle = decodeURIComponent(heroTitle.replace(/\+/g, ' '));
                const decodedHeroSubtitle = decodeURIComponent(heroSubtitle.replace(/\+/g, ' '));
                const decodedAgentName = decodeURIComponent(agentName.replace(/\+/g, ' '));
                const decodedAgentTitle = decodeURIComponent(agentTitle.replace(/\+/g, ' '));
                const decodedAgentIframeUrl = decodeURIComponent(agentIframeUrl);
                const decodedProductTitle = decodeURIComponent(productTitle.replace(/\+/g, ' '));
                const decodedProductDescription = decodeURIComponent(productDescription.replace(/\+/g, ' '));
                const decodedFinalCtaTitle = decodeURIComponent(finalCtaTitle.replace(/\+/g, ' '));

                // Update hidden data elements
                document.getElementById('company-name').textContent = decodedCompanyName;
                document.getElementById('hero-title').textContent = decodedHeroTitle;
                document.getElementById('hero-subtitle').textContent = decodedHeroSubtitle;
                document.getElementById('agent-name').textContent = decodedAgentName;
                document.getElementById('agent-title').textContent = decodedAgentTitle;
                document.getElementById('agent-iframe-url').textContent = decodedAgentIframeUrl;
                document.getElementById('agent-iframe').src = decodedAgentIframeUrl;
                document.getElementById('product-title').textContent = decodedProductTitle;
                document.getElementById('product-description').textContent = decodedProductDescription;
                document.getElementById('finalCta-title').textContent = decodedFinalCtaTitle;

            }
        }

        // Lägg bara till detta i slutet:
        window.addEventListener('load', parseURLParameters);
        window.addEventListener('hashchange', parseURLParameters);

        // Kör direkt om sidan redan är laddad
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', parseURLParameters);
        } else {
            parseURLParameters();
        }
