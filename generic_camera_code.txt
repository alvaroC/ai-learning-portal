    <script>

        let currentStream = null;
        let lastAnalysisResult = null;
        
        // Initialize camera when page loads
        window.addEventListener('load', initializeCamera);
        
        async function initializeCamera() {
            try {
                currentStream = await navigator.mediaDevices.getUserMedia({
                    video: { 
                        facingMode: 'environment',
                        width: { ideal: 1920 },
                        height: { ideal: 1080 }
                    }
                });
                
                const video = document.getElementById('camera-preview');
                video.srcObject = currentStream;
                await video.play();
                
                // Show camera interface
                document.getElementById('no-camera').style.display = 'none';
                document.getElementById('camera-preview').style.display = 'block';
                document.getElementById('camera-overlay').style.display = 'block';
                document.getElementById('camera-controls').style.display = 'flex';
                
            } catch (error) {
                console.error('Kamerinitialisering misslyckades:', error);
                // Keep upload option visible
            }
        }
        
        function capturePhoto() {
            const video = document.getElementById('camera-preview');
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            
            canvas.width = video.videoWidth;
            canvas.height = video.videoHeight;
            ctx.drawImage(video, 0, 0);
            
            // Show analysis overlay
            document.getElementById('analysis-overlay').style.display = 'flex';
            
            // Convert to blob and analyze
            canvas.toBlob(blob => {
                analyzeImage(blob);
            }, 'image/jpeg', 0.85);
        }
        
        function handleFileUpload(input) {
            const file = input.files[0];
            if (file) {
                document.getElementById('analysis-overlay').style.display = 'flex';
                analyzeImage(file);
            }
        }

        async function analyzeImage(imageBlob) {
            try {
                const formData = new FormData();
                formData.append('data', imageBlob, 'captured_image.jpg');
                formData.append('timestamp', new Date().toISOString());
                formData.append('analysis_type', 'generic');
                
                // Use your actual webhook URL                
                const response = await fetch('https://ai-business-lab-n8n.u7ysvb.easypanel.host/webhook/webhook-camera', {
                    method: 'POST',
                    body: formData
                });
                
                if (!response.ok) {
                    throw new Error(`HTTP fel! status: ${response.status}`);
                }
                
                const result = await response.json();
                console.log('n8n Svar:', result);
                
                // Hide analysis overlay
                document.getElementById('analysis-overlay').style.display = 'none';
                
                if (result.success && result.analysis) {
                    // Store the result for later use
                    lastAnalysisResult = result.analysis;
                    
                    // Parse the analysis (it might be a JSON string)
                    let analysisData;
                    try {
                        analysisData = typeof result.analysis === 'string' 
                            ? JSON.parse(result.analysis) 
                            : result.analysis;
                    } catch (e) {
                        // If parsing fails, treat as plain text
                        analysisData = {
                            imageType: 'ok√§nd',
                            description: result.analysis,
                            confidence: 'ok√§nd'
                        };
                    }
                    
                    displayResults(analysisData);
                } else {
                    throw new Error(result.error || 'Analys misslyckades - inga resultat returnerades');
                }
                
            } catch (error) {
                console.error('Bildanalys misslyckades:', error);
                document.getElementById('analysis-overlay').style.display = 'none';
                
                // Show error in results overlay
                displayResults({
                    imageType: 'fel',
                    description: `Analys misslyckades: ${error.message}`,
                    confidence: 0
                }, true);
            }
        }

        function displayResults(analysisData, isError = false) {
            const resultsOverlay = document.getElementById('results-overlay');
            const resultsText = document.getElementById('results-text');
            const resultsIcon = document.getElementById('results-icon');
            
            if (isError) {
                resultsIcon.textContent = '‚ùå';
                resultsText.innerHTML = `
                    <strong>Fel:</strong><br>
                    ${analysisData.description}
                `;
            } else {
                // Set appropriate icon based on image type
                const iconMap = {
                    'tak': 'üè†',
                    'roof': 'üè†',
                    'dokument': 'üìÑ',
                    'document': 'üìÑ',
                    'utrustning': '‚öôÔ∏è',
                    'equipment': '‚öôÔ∏è',
                    'diagram': 'üìä',
                    'chart': 'üìä',
                    'fastighet': 'üè°',
                    'property': 'üè°',
                    'annat': 'üñºÔ∏è',
                    'other': 'üñºÔ∏è',
                    'ok√§nd': 'ü§ñ',
                    'unknown': 'ü§ñ'
                };
                
                resultsIcon.textContent = iconMap[analysisData.imageType] || 'ü§ñ';
                
                resultsText.innerHTML = `
                    <strong>Typ:</strong> ${analysisData.imageType || 'Ok√§nd'}<br>
                    ${analysisData.confidence ? `<strong>S√§kerhet:</strong> ${analysisData.confidence}%<br>` : ''}
                    <strong>Analys:</strong><br>
                    ${analysisData.description || analysisData.analysis?.description || 'Ingen beskrivning tillg√§nglig'}
                `;
            }
            
            resultsOverlay.style.display = 'flex';
        }

        function sendToChat() {
            if (lastAnalysisResult && window.parent) {
                // Format the message for Emma
                let message = '';
                
                try {
                    const analysisData = typeof lastAnalysisResult === 'string' 
                        ? JSON.parse(lastAnalysisResult) 
                        : lastAnalysisResult;
                    
                    message = `üì∏ Bildanalysresultat:\n\n` +
                             `Typ: ${analysisData.imageType || 'Ok√§nd'}\n` +
                             (analysisData.confidence ? `S√§kerhet: ${analysisData.confidence}%\n` : '') +
                             `Beskrivning: ${analysisData.description || analysisData.analysis?.description || 'Ingen beskrivning tillg√§nglig'}`;
                } catch (e) {
                    message = `üì∏ Bildanalysresultat:\n\n${lastAnalysisResult}`;
                }
                
                // Send to parent window (main chat interface)
                window.parent.postMessage({
                    type: 'emma-message',
                    message: message
                }, '*');
                
                // Close the camera
                closeCamera();
            } else {
                alert('Inga analysresultat att skicka!');
            }
        }

        function takeAnother() {
            // Hide results and show camera again
            document.getElementById('results-overlay').style.display = 'none';
            lastAnalysisResult = null;
        }

        function closeCamera() {
            // Send close message to parent
            if (window.parent) {
                window.parent.postMessage({
                    type: 'close-camera'
                }, '*');
            }
        }

        // Cleanup when page unloads
        window.addEventListener('beforeunload', () => {
            if (currentStream) {
                currentStream.getTracks().forEach(track => track.stop());
            }
        });

        // Handle messages from parent window
        window.addEventListener('message', function(event) {
            if (event.data && event.data.type === 'close-camera') {
                if (currentStream) {
                    currentStream.getTracks().forEach(track => track.stop());
                }
            }
        });

</script>
